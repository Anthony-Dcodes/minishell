typedef struct s_listex
{
	char		**items;
	char		*xattr_qc;
	char		*xattr_pc;
	char		*xattr_sc;
	char		*xattr_pidx;
	char		*xattr_rc;
	size_t		size;
}	t_listex;

int	ft_parsecmd(t_s *s, char *src, t_listex *dst)
{
	t_listex		parsed_quotes;
	t_listex		**parsed_pipes;
	t_listex		**parsed_exp;
	int				r;

	ft_parsequotes2(s, &parsed_quotes, src)
	-----ft_detectchargroups2(s, src)
	-----ft_createchargroups2(s, src)
	-----ft_parsesinglequotes2(s)
	-----ft_parsedoublequotes2(s)
	-----ft_copyandclean2(s, dst)
	ft_parsepipes2(s, &parsed_pipes, &parsed_quotes)
	ft_parseexp2(s, &parsed_exp, parsed_pipes)
	ft_preparelist2(s, dst, parsed_exp)
	ft_clean(s, &parsed_quotes, parsed_pipes, parsed_exp)
	return (SUCCESS);
}
